*Calculator.js

import { LightningElement } from 'lwc';

export default class CalculatorLWC extends LightningElement {

    num1;
    num2;
    result;

    valueNum1(event) {
        this.num1 = event.target.value;
    }
    valueNum2(event) {
        this.num2 = event.target.value;
    }

    add() {
        this.result = Number(this.num1) + Number(this.num2);
    }

    sub() {
        this.result = Number(this.num1) - Number(this.num2);
    }

    mul() {
        this.result = Number(this.num1) * Number(this.num2);
    }

    div() {
        this.result = Number(this.num1) / Number(this.num2);
    }
}
